25#define pb(x) push_back(x)

27#define MX 300005
28const uint64_t MOD = 998244353;
29int n, m, tot;
30int rts[MX];
31struct seg {
32 int l, r;
33 uint64_t m, p;
34} a[1 << 22];
35
36inline void pushup(const int &p) { a[p].p = (a[a[p].l].p + a[a[p].r].p) % MOD; }
37
38inline void pushdown(const int &p) {
39 if (a[p].m == 1)
40 return;
41 a[a[p].l].p = a[a[p].l].p * a[p].m % MOD;
42 a[a[p].r].p = a[a[p].r].p * a[p].m % MOD;
43 a[a[p].l].m = a[a[p].l].m * a[p].m % MOD;
44 a[a[p].r].m = a[a[p].r].m * a[p].m % MOD;
45 a[p].m = 1;
46}
47
48uint64_t query(int p, int pos) {
49 int l = 0, r = m;
50 while (r - l > 1) {
51 pushdown(p);
52 int mid = l + (r - l) / 2;
53 if (pos < mid)
54 p = a[p].l, r = mid;
55 else
56 p = a[p].r, l = mid;
57 }
58 return a[p].p;
59}
60
61void insert(int &ori, int pos, uint64_t val) {
62 ori = ++tot;
63 int p = ori;
64 int l = 0, r = m;
65 while (r - l > 1) {
66 a[p].m = 1;
67 a[p].p = val;
68 int mid = l + (r - l) / 2;
69 if (pos < mid)
70 p = a[p].l = ++tot, r = mid;
71 else
72 p = a[p].r = ++tot, l = mid;
73 }
74 a[p].p = val;
75 a[p].m = 1;
76}
77
78uint64_t pr, sl, sr, cl, cr;
79void merge(int q, int &p) {
80 if (p == q && q == 0)
81 return;
82 if (q == 0) {
83 uint64_t v = pr * sl % MOD + (1 + MOD - pr) * sr % MOD;
84 v %= MOD;
85 a[p].p = a[p].p * v % MOD;
86 a[p].m = a[p].m * v % MOD;
87 return;
88 }
89 if (p == 0) {
90 uint64_t v = pr * cl % MOD + (1 + MOD - pr) * cr % MOD;
91 v %= MOD;
92 a[q].p = a[q].p * v % MOD;
93 a[q].m = a[q].m * v % MOD;
94 p = q;
95 return;
96 } // No same value'll appear in both child.
97 pushdown(p);
98 pushdown(q);
99 uint64_t t1 = a[a[q].l].p + sl, t2 = a[a[p].l].p + cl;
100 uint64_t t3 = a[a[q].r].p + sr, t4 = a[a[p].r].p + cr;
101 swap(sr, t3), swap(cr, t4);
102 merge(a[q].l, a[p].l);
103 swap(sr, t3), swap(cr, t4);
104 swap(sl, t1), swap(cl, t2);
105 merge(a[q].r, a[p].r);
106 swap(sl, t1), swap(cl, t2);
107 pushup(p);
108}

110vector<int> G[MX];
111int c[MX];
112pair<uint64_t, int> b[MX];
113uint64_t Pr[MX];
114
115void dfs(int u) {
116 if (G[u].empty()) {
117 insert(rts[u], c[u], 1);
118 } else {
119 for (auto i : G[u]) {
120 dfs(i);
121 }
122 rts[u] = rts[G[u][0]];
123 if (G[u].size() == 2) {
124 pr = Pr[u];
125 merge(rts[G[u][1]], rts[u]);
126 }
127 }
128}

130void solve() {
131 cin >> n;
132 for (int i = 1; i <= n; ++i) {
133 int tmp;
134 cin >> tmp;
135 G[tmp].push_back(i);
136 }
137 for (int i = 1; i <= n; ++i) {
138 uint64_t tmp;
139 cin >> tmp;
140 if (G[i].empty()) b[m] = make_pair(tmp, i), ++m;
141 else Pr[i] = tmp * 796898467ull % MOD;
142 }
143 sort(b, b + m);
144 for (int i = 0; i < m; ++i) {
145 c[b[i].second] = i;
146 }
147 dfs(1);
148 uint64_t ans = 0;
149 for (int i = 0; i < m; ++i) {
150 uint64_t t = query(rts[1], i);
151 ans += (i + 1) * b[i].first % MOD * t % MOD * t % MOD;
152 ans %= MOD;
153 }
154 cout << ans << endl;
155}

157int main() {

159 while (T--)
160 solve();
161 return 0;
