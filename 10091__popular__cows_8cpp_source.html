18int head[MX], to[MAX_M], nxt[MAX_M], tot;
19int scc_id[MX], cnt;
20int low[MX], dfn[MX], out[MX], siz[MX];
21bool in_stack[MX];
22
23stack<int> s;
24int t;
25void tarjan(int u) {
26 dfn[u] = low[u] = ++t;
27 in_stack[u] = true;
28 s.push(u);
29 for (int i = head[u]; i; i = nxt[i]) {
30 int &v = to[i];
31 if (!low[v]) {
32 tarjan(v);
33 low[u] = min(low[u], low[v]);
34 } else if (in_stack[v]) low[u] = min(low[u], dfn[v]);
35 }
36 if (low[u] == dfn[u]) {
37 ++cnt;
38 int v;
39 do {
40 siz[cnt] ++;
41 v = s.top();
42 s.pop();
43 scc_id[v] = cnt;
44 in_stack[v] = false;
45 } while (v != u);
46 }
47}
