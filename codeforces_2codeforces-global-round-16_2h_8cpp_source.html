52const int N = 1000005;
53int c, n;
54int a[N], low[N], w[N], pri[N], tot;
55u32 f[N], g[N], gg[N];
56vector<int> grp[N];

58void prework() {
59 a[1] = 1; w[1] = 1;
60 FOR(i, 2, c + 1) {
61 if (!low[i]) {
62 low[i] = i;
63 w[i] = i;
64 a[i] = 1;
65 pri[tot++] = i;
66 }
67 for (int j = 0; j < tot && i * pri[j] <= c; ++j) {
68 if (pri[j] < low[i]) {
69 low[i * pri[j]] = pri[j];
70 a[i * pri[j]] = a[i];
71 w[i * pri[j]] = w[i] * pri[j];
72 } else {
73 low[i * pri[j]] = pri[j];
74 a[i * pri[j]] = 0;
75 w[i * pri[j]] = w[i];
76 break;
77 }
78 }
79 }
80 FOR(i, 1, c + 1) {
81 grp[w[i]].pb(i);
82 }
83 int cnt = 0;
84 static int ask[N];
85 FOR(i, 1, c + 1) {
86 if (a[i] == 1) {
87 ask[cnt++] = i;
88 }
89 }
90 cout << cnt << ' ';
91 FOR(i, 0, cnt) cout << ask[i] << ' ';
92 cout << endl;
93 FOR(i, 0, cnt) {
94 cin >> g[ask[i]];
95 }
96 int total_xor = g[1];
97 for (int x = 1; x <= c; x++) {
98 if (grp[x].empty()) continue;
99 for (int y = x; y <= c; y += x) f[y] ^= (g[x] ^ total_xor);
100 }
101 f[1] = total_xor;
102 for (int x = c; x >= 1; x--) {
103 if (grp[x].empty()) continue;
104 for (int y = 2 * x; y <= c; y += x) {
105 if (grp[y].size()) f[x] ^= f[y];
106 }
107 }
108}

116const int rand_tot = 500;
117u32 s[N];
118void gauss(int x) {
119 int n = grp[x].size();
120 FOR(i, 0, n) s[i] = grp[x][i], p[i] = 0, p[i].set(i);
121 int i = 0;
122 for (int i = 0; i < n; ++i) {
123 int f = -1;
124 for (int j = 0; j < n && f < 0; ++j)
125 if ((s[j] & -s[j]) == 1 << i) f = j;
126 if (f == -1) continue;
127 for (int j = 0; j < n; ++j) {
128 if (s[j] & (1 << i) && j != f) s[j] ^= s[f], p[j] ^= p[f];
129 }
130 }
131 int all = 0;
132 int cnt = 0;
133 bitset<260> to;
134 for (int i = 0; i < n; ++i) {
135 if (s[i]) ++cnt;
136 if (s[i] & f[x]) {
137 to ^= p[i];
138 }
139 }
140 if (cnt == n) {
141 FOR(i, 0, n) {
142 if (to[i]) ans.pb(grp[x][i]);
143 }
144 return;
145 }
146 vector<vector<int> > cgrp;
147 vector<int> csiz;
148 vector<bool> vis(260);
149 FOR(_, 0, rand_tot) {
150 bitset<260> cur = to;
151 FOR(i, 0, n) {
152 if (!s[i] && gen() & 1) {
153 cur ^= p[i];
154 }
155 }
156 if (!vis[cur.count()]) {
157 vector<int> cans;
158 FOR(i, 0, n) {
159 if (cur[i]) cans.pb(grp[x][i]);
160 }
161 cgrp.pb(cans);
162 csiz.pb(cur.count());
163 vis[cur.count()] = 1;
164 }
165 }
166 ext.pb(cgrp);
167 szext.pb(csiz);
168}

192 n -= szext[i-1][j];
193 for (auto &e : ext[i-1][j]) {
