4#include <algorithm>
5using std::swap;

7/**********************************************************************************************************
8 * LCT template interface:
9 * const int N: maximum size of forest
10 * class attr: extra attributes & lazy tags to be maintained
11 * bool attr::active(): whether the lazy tag has effect or not.
12 * void attr::reset(): reset lazy tag to be inactive.
13 * void attr::apply(const attr&): push lazy tags to current node.
14 * void attr::merge(const attr&, const attr&): merge attributes of current node and it's childrens.
15 *********************************************************************************************************/
16
17template<int N, class attr>
18struct LCT {
19 struct node {
20 int ch[2], f;
21 bool flp;
22 attr var;// extra attributes & tags here
23 inline void _flp() {
24 swap(ch[0], ch[1]), flp ^= 1;
25 }
26 } a[N];
27#define lc(p) a[p].ch[0]
28#define rc(p) a[p].ch[1]
29#define f(p) a[p].f
30 inline bool typ(int p) {
31 return p == rc(f(p));
32 }
33 inline bool isroot(int p) { // chk root of splay
34 return !(lc(f(p)) == p || rc(f(p)) == p);
35 }
36 inline void up(int p) {
37 a[p].var.merge(a[lc(p)].var, a[rc(p)].var); // maintain extra attributes
38 }
39 inline void down(int p) {
40 if (a[p].flp) {
41 if (lc(p)) a[lc(p)]._flp();
42 if (rc(p)) a[rc(p)]._flp();
43 a[p].flp = 0;
44 }
45 if (!a[p].var.active()) {
46 if (lc(p)) a[lc(p)].var.apply(a[p].var);
47 if (rc(p)) a[rc(p)].var.apply(a[p].var);
48 a[p].var.reset();
49 } // maintain extra tags
50 }
51 void upd(int p) { // clean tags along the path
52 if (!isroot(p)) upd(f(p));
53 down(p);
54 }
55 inline void rotate(int x) { // modified splay rotate
56 int o = typ(x), y = f(x), z = f(y);
57 if (!isroot(y)) {
58 a[z].ch[typ(y)] = x;
59 }
60 a[a[x].ch[o^1]].f = y;
61 a[y].ch[o] = a[x].ch[o^1];
62 a[x].ch[o^1] = y;
63 a[y].f = x;
64 a[x].f = z;
65 up(y);
66 up(x);
67 }
68 inline void splay(int p) {
69 upd(p);
70 for (int x; !isroot(p); rotate(p)) {
71 x = f(p);
72 if (!isroot(x)) rotate((typ(p) == typ(x)) ? x : p);
73 }
74 }
75 inline int access(int p) { // make a splay out of path root -> p
76 int lst = 0;
77 for (; p; lst = p, p = f(p)) {
78 splay(p);
79 a[p].ch[1] = lst;
80 up(p);
81 }
82 return lst;
83 }
84 inline int getrt(int p) { // root of real tree (will also access(p))
85 p = access(p);
86 while (lc(p)) p = lc(p);
87 splay(p);
88 return p;
89 }
90 inline void setrt(int p) { // make p the root of it's component
91 p = access(p);
92 a[p]._flp();
93 }
94 inline void link(int u, int v) {
95 setrt(u);
96 if (getrt(v) != u) {
97 splay(u);
98 f(u) = v;
99 }
100 }
101 inline void cut(int u, int v) {
102 setrt(u);
103 if (getrt(v) == u && f(v) == u && lc(v) == 0) {
104 f(v) = rc(u) = 0;
105 up(u);
106 }
107 }
108 inline void path(int u, int v) { // make a splay out of path u -> v, and set v as the root of it.
109 setrt(u);
110 access(v);
111 splay(v); // cost amortized
112 }
113 attr &operator[] (int i) {
114 return a[i].var;
115 }
116 const attr &operator[] (int i) const {
117 return a[i].var;
118 }
119};
