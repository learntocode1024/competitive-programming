32int rd() {
33 unsigned u = 0, s = 0;
34 while (*o && *o <= 32) ++o; // skip whitespaces...
35 if (*o == '-')
36 s = ~s, ++o;
37 else if (*o == '+')
38 ++o; // skip sign
39 while (*o >= '0' && *o <= '9')
40 u = (u << 3) + (u << 1) + (*o++ - '0'); // u * 10 = u * 8 + u * 2 :)
41 return static_cast<int>((u ^ s) + !!s);
42}

75using IO::rd;

77const int N = 1e5+5;
78int n, m;
79vector<int> G[N];
80vector<int> f[N];
81int dep[N], dfn[N], rnk[N], rmq[21][N << 1], tot, _fa[N], tot2, dsn[N];
82void dfs(int u, int fa) {
83 dfn[u] = ++tot;
84 dsn[u] = ++tot2;
85 rnk[tot2] = u;
86 _fa[u] = fa;
87 dep[u] = dep[fa] + 1;
88 rmq[0][tot] = u;
89 for (auto v : G[u]) {
90 if (v != fa) dfs(v, u), rmq[0][++tot] = u;
91 }
92}
93inline int gmin(int a, int b) {
94 return dep[a] < dep[b] ? a : b;
95}
96inline void init() {
97 // dep[0] = -1;
98 dfs(1, 0);
99 dep[0] = 0;
100 FOR(i, 1, 21) {
101 for (int j = 1; j + (1 << i) - 1 <= tot; ++j) {
102 rmq[i][j] = gmin(rmq[i - 1][j], rmq[i - 1][j + (1 << (i - 1))]);
103 }
104 }
105}
106int lca(int a, int b) {
107 int x = dfn[a], y = dfn[b];
108 if (x > y) swap(x, y);
109 int t = 31 - __builtin_clz(y - x + 1);
110 return gmin(rmq[t][x], rmq[t][y + 1 - (1 << t)]);
111}

113struct DS {
114 struct node {
115 int _lc, _rc;
116 int l, r, s, u;
117 int cnt;
118 } a[N << 5];
119 int all;
120 #define lc a[a[p]._lc]
121 #define rc a[a[p]._rc]
122 #define P a[p]
123 inline void up(int p) {
124 if (lc.l && rc.l) {
125 P.l = lc.l;
126 P.r = rc.r;
127 P.s = lc.s + rc.s - dep[lca(rnk[lc.r], rnk[rc.l])];
128 P.u = lca(lc.u, rc.u);
129 } else {
130 P.l = lc.l | rc.l;
131 P.r = lc.r | rc.r;
132 P.s = lc.s | rc.s;
133 P.u = lc.u | rc.u;
134 }
135 }
136 void merge(int &p, int q, int l, int r) { // Even segtree merge operation needs l, r arguments (to avoid leaves being overwrited)
137 if (!p || !q) {
138 p |= q;
139 return;
140 }
141 if (r - l == 1) {
142 a[p].cnt += a[q].cnt;
143 a[p].l |= a[q].l;
144 a[p].r |= a[q].r;
145 a[p].u |= a[q].u;
146 a[p].s |= a[q].s;
147 return;
148 }
149 int mid = (l + r) >> 1;
150 merge(a[p]._lc, a[q]._lc, l, mid);
151 merge(a[p]._rc, a[q]._rc, mid, r);
152 up(p);
153 }
154 void ins(int &p, int l, int r, int pos, int v) {
155 if (!p) p = ++all;
156 if (r - l == 1) {
157 P.cnt += v;
158 if (P.cnt > 0) {
159 P.l = P.r = l;
160 P.u = rnk[l];
161 P.s = dep[rnk[l]];
162 } else {
163 P.l = P.r = P.s = P.u = 0;
164 }
165 return;
166 }
167 int mid = (l + r) >> 1;
168 if (pos < mid) ins(a[p]._lc, l, mid, pos, v);
169 else ins(a[p]._rc, mid, r, pos, v);
170 up(p);
171 }
172 int get(int rt) {
173 return a[rt].s - dep[a[rt].u];
174 }
175 #undef lc
176 #undef rc
177 #undef P
178} T;

180int rts[N];
181i64 ans;
182
183void calc(int u, int fa) {
184 for (auto v : G[u]) {
185 if (v != fa) {
186 calc(v, u);
187 T.merge(rts[u], rts[v], 1, n + 1);
188 }
189 }
190 for (auto v : f[u]) {
191 if (v > 0) {
192 T.ins(rts[u], 1, n + 1, v, 1);
193 } else {
194 T.ins(rts[u], 1, n + 1, -v, -1);
195 }
196 }
197 ans += T.get(rts[u]);
198 // cerr << "clac(" << u << ") = " << T.get(rts[u]) << '\n';
199}

201void solve() {
202 n = rd(), m = rd();
203 FOR(i, 1, n) {
204 int u = rd(), v = rd();
205 G[u].pb(v);
206 G[v].pb(u);
207 }
208 init();
209 FOR(i, 0, m) {
210 int s = rd(), t = rd();
211 if (s == t) continue;
212 int x = dsn[s], y = dsn[t], z = lca(s, t);
213 f[s].push_back(x);
214 f[s].push_back(y);
215 f[t].push_back(x);
216 f[t].push_back(y);
217 f[z].push_back(-x);
218 f[z].push_back(-y);
219 f[_fa[z]].push_back(-x);
220 f[_fa[z]].push_back(-y);
221 }
222 calc(1, 0);
223 cout << ans / 2 << '\n';
224}
